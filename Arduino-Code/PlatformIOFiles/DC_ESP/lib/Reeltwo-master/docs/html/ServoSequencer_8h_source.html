399 uint16_t speedMinMS, uint16_t speedMaxMS, float startPos = 0.0, float endPos = 1.0,
400 float (*onEasingMethod)(float) = NULL, float (*offEasingMethod)(float) = NULL)
401 {
402 stop();

404 fServoGroupMask = servoGroupMask;
405 fLength = length;
406 fStartPos = startPos;
407 fEndPos = endPos;

409 fSpeedMaxMS = speedMaxMS;

419 uint16_t speedMS = 0, float startPos = 0.0, float endPos = 1.0)
420 {
421 playVariableSpeed(sequence, length, servoGroupMask, speedMS, speedMS, startPos, endPos);

424 void stop()
425 {
426 fSequence = NULL;
427 fLength = 0;
428 fNextStepMS = 0;
429 }
430
431 virtual void animate() override
432 {
433 unsigned long currentTime;
434 if (fSequence == NULL || (currentTime = millis()) < fNextStepMS)
435 return;
436 if (fIndex >= fLength)
437 {
438 stop();
439 return;
440 }
441 uint32_t servoSetMask = 0;
442 const ServoStep* step = &fSequence[fIndex];
443 while (pgm_read_word(&step->cs) == uint16_t(~0))
444 {
445 // Check for an initial range limiting step
446 uint16_t offsetFromStart = 0;
447 uint16_t offsetFromEnd = 0;
448 offsetFromStart |= (uint32_t)pgm_read_byte(&step->servo1_8) << 8;
449 offsetFromStart |= (uint32_t)pgm_read_byte(&step->servo9_16);
450 offsetFromEnd |= (uint32_t)pgm_read_byte(&step->servo17_24) << 8;
451 offsetFromEnd |= (uint32_t)pgm_read_byte(&step->servo25_32);
452 fOffsetFromStart = (offsetFromStart / 1000.0);
453 fOffsetFromEnd = (offsetFromEnd / 1000.0);
454 fIndex++;
455 if (fIndex >= fLength)
456 {
457 stop();
458 return;
459 }
460 step = &fSequence[fIndex];
461 }
462 servoSetMask |= (uint32_t)pgm_read_byte(&step->servo1_8) << 24;
463 servoSetMask |= (uint32_t)pgm_read_byte(&step->servo9_16) << 16;
464 servoSetMask |= (uint32_t)pgm_read_byte(&step->servo17_24) << 8;
465 servoSetMask |= (uint32_t)pgm_read_byte(&step->servo25_32) << 0;
466 uint32_t ms = pgm_read_word(&fSequence[fIndex].cs) * 10L;
467 fNextStepMS = currentTime + fSpeedMinMS + ms;
468 fDispatch.moveServoSetTo(fServoGroupMask, servoSetMask, 0,
469 fSpeedMinMS, fSpeedMaxMS, fEndPos-fOffsetFromEnd, fStartPos+fOffsetFromStart, fOnEasingMethod, fOffEasingMethod);
470 fIndex++;
471 }
472
473private:
474 ServoDispatch& fDispatch;
475 ServoStep* fSequence;
476 uint16_t fLength = 0;
477 uint16_t fIndex = 0;
478 float fOffsetFromStart;
479 float fOffsetFromEnd;
480 float fStartPos;
481 float fEndPos;
482 uint16_t fSpeedMinMS;
483 uint16_t fSpeedMaxMS;
484 uint32_t fServoGroupMask;
485 uint32_t fNextStepMS;
486 float (*fOnEasingMethod)(float);
487 float (*fOffEasingMethod)(float);
488};
