235 virtual void handleCommand(const char* cmd) override
236 {
237 int durationSec = -1;
238 byte typeState = 0;
239 byte functionState = 0;
240 int optionState = -1;
241 int optionState2 = -1;
242
243 if (*cmd++ != 'H' || *cmd++ != 'P')
244 return;
245
246 int commandLength = strlen(cmd);
247 const char* pipeIndex = strchr(cmd, '|');
248 if (pipeIndex != NULL)
249 {
250 durationSec = atoi(pipeIndex+1);
251 commandLength = pipeIndex - cmd;
252 }
253 if (cmd[0]=='F' || // Front HP
254 cmd[0]=='R' || // Rear HP
255 cmd[0]=='T' || // Top HP
256 cmd[0]=='X' || // Front & Rear HPs
257 cmd[0]=='Y' || // Front & Top HPs
258 cmd[0]=='Z' || // Rear & Top HPs
259 cmd[0]=='D' || // Radar eye
260 cmd[0]=='O' || // Other HP
261 cmd[0]=='A' ) // All three HP
262 {
263 if (commandLength >= 2)
264 typeState = cmd[1]-'0';
265 if (commandLength >= 4)
266 functionState = (cmd[2]-'0')*10+(cmd[3]-'0');
267
268 if (commandLength >= 5)
269 {
270 optionState = cmd[4]-'0';
271 }
272 else if (typeState == 1)
273 {
274 optionState = 1; // Set Center as the fallback position
275 }
276 if (commandLength >= 6)
277 {
278 optionState2 = cmd[5]-'0';
279 }
280 // All match A
281 bool match = (cmd[0]=='A');
282 // Front holo matches FXY
283 match = (match || (fID == kFrontHolo && (cmd[0]=='F' || cmd[0]=='X' || cmd[0]=='Y')));
284 // Rear holo matches RXZ
285 match = (match || (fID == kRearHolo && (cmd[0]=='R' || cmd[0]=='X' || cmd[0]=='Z')));
286 // Top holo matches TYZ
287 match = (match || (fID == kTopHolo && (cmd[0]=='T' || cmd[0]=='Y' || cmd[0]=='Z')));
288 // Radar eye matches D
289 match = (match || (fID == kRadarEye && cmd[0]=='D'));
290 // Other holo matches D
291 match = (match || (fID == kOtherHolo && cmd[0]=='O'));
292 if (match)
293 {
294 if (typeState == 0)
295 {
296 flushLEDState(); // Flushes LED Command Arrays
297 fLEDFunction = functionState;
298 if (optionState >= 0)
299 {
300 if (optionState == 0)
301 fLEDOption1 = random(0,9);
302 else
303 fLEDOption1 = optionState;
304 }
305 else
306 {
307 if (functionState == 7)
308 {
309 fLEDOption1 = shortColor;
310 }
311 else
312 fLEDOption1 = defaultColor;
313 }
314 if (optionState2 >= 0 && functionState == 3)
315 fLEDOption2 = optionState2;
316 else if (optionState2 < 0 && functionState == 3)
317 fLEDOption2 = kDimPulseSpeed;
318 fLEDHalt = (durationSec >= 1) ? durationSec : 0;
319 fLEDHaltTime = millis();
320 varResets();
321 }
322 else if (typeState == 1)
323 {
324 flushHPState(); // Flushes HP Command Array
325 fHPFunction = functionState;
326 fHPOption1 = optionState;
327 if (durationSec >= 1)
328 fHPHalt = durationSec;
329
330 fHPHaltTime = millis();
331 fWagCount = -1;
332 }
333 }
334 }
335 else if (cmd[0] == 'S') // Major Sequences (Both LEDS and HP Servos)
336 {
337 if (commandLength >= 2)
338 {
339 functionState = (cmd[1]-'0');
340 selectSequence(functionState, durationSec);
341 }
342 }

518 virtual void selectSequence(int sequence, int durationSec)
519 {
520 switch (sequence)
521 {
522 case 1:
523 varResets(); // Resets the Variables for LEDs
524 flushLEDState(); // Flushes LED Command Arrays
525 fHPFunction = 0; // Set HP to Do Nothing
526 fLEDFunction = 0; // Set LED to Do Nothing
527 if (fID == kFrontHolo)
528 {
529 moveHP(kDown, SERVO_SPEED[0]); // Moves Front HP to Down Position
530 fLEDFunction = 1; // Set Leia LED Sequence to run each loop.
531 }
532 if (durationSec >= 1)
533 {
534 fLEDHalt = durationSec; // Set LED Halt timeout if passed via command.
535 fLEDHaltTime = millis(); // and note start time
536 fHPHalt = durationSec; // Set HP Halt timeout if passed via command.
537 fHPHaltTime = millis(); // and note start time
538 }
539 break;
540 case 7:
541 fEnableTwitchHP = false; // Disables Auto HP Twith
542 flushLEDState(); // Flushes LED Command Array
543 flushHPState(); // Flushes HP Command Array
544 off(); // Turns Off LEDs
545 fEnableTwitchLED = 0; // Disables Auto LED Twith
546 break;
547 case 8:
548 fEnableTwitchHP = true; // Enables Auto HP Twith
549 flushLEDState(); // Flushes LED Command Array
550 flushHPState(); // Flushes HP Command Array
551 off(); // Turns Off LEDs
552 fEnableTwitchLED = 1; // Enables Auto LED Twith
553 break;
554 case 9:
555 fEnableTwitchHP = true; // Enables Auto HP Twith on all HPs
556 flushLEDState(); // Flushes LED Command Array
557 flushHPState(); // Flushes HP Command Array
558 off(); // Turns Off LEDs
559 fEnableTwitchLED = 2; // Enables Auto LED Twith on all HPs using random sequences
560 break;
561 default:
562 break;
563 }

841 {
842 uint16_t numLEDs = numPixels();
843 const unsigned int inter = 75;
844 if ((millis() - fCounter) > inter)
845 {
846 fCounter = millis();
847 if (numLEDs == 7)
848 {
849 /* 7 pixels with one center */
850 setPixelColor(0, kOff); // Center LED is always off
851 if (fFrame >= numLEDs)

853 fFrame = 0;
854 }
855 for (unsigned i = 1; i < numLEDs; i++)
856 {
857 if (i == fFrame)

861 else
862 {
863 setPixelColor(i, kOff);
864 }
865 }
866 }
867 else
868 {
869 /* More than 7 pixels no center */
870 if (fFrame >= numLEDs)
871 {
872 fFrame = 0;
873 }
874 for (unsigned i = 0; i < numLEDs; i++)
875 {
876 if (i == fFrame)
877 {
878 setPixelColor(i, basicColor(c));
879 }
880 else
881 {
882 setPixelColor(i, kOff);
883 }
884 }
885 }
