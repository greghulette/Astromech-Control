1203 //function to calculate all colors based on the chosen colorPalNum
1204 //this will get called during setup(), and also if we've remotely commanded to change color palettes
1205 //NEW: added a "bright" parameter, to allow front and rear brightness to be adjusted independantly
1206 void calculateAllColors(byte colorPalNum, byte brightVal)
1207 {
1208 //we define both all palettes, in case the user wants to try out rear colors on the front etc
1209 // note that these are not RGB colors, they're HSV
1210 // for help calculating HSV color values see http://joymonkey.com/logic/
1211 static const HSVColor keyColors[PAL_COUNT][4] PROGMEM = {
1212 { {170, 255, 0} , {170, 255, 85} , {170, 255, 170} , {170, 0, 170} } , //front colors
1213 { { 90, 235, 0} , { 75, 255, 250} , { 30, 255, 184} , { 0, 255, 250} } , //rear colors (hues: 87=bright green, 79=yellow green, 45=orangey yellow, 0=red)
1214 { { 0, 255, 0} , { 0, 255, 0} , { 0, 255, 100} , { 0, 255, 250} } , //monotone (black to solid red)
1215 { { 0, 255, 0} , { 0, 255, 250} , { 40, 255, 0} , { 40, 255, 250} } , //dual color red and yellow
1216 { {165, 50, 248} , {166, 181, 226} , {165, 223, 89} , {255, 255, 214} } , //blue and red
1217 { {87, 206, 105} , { 79, 255, 214} , { 43, 255, 250} , { 25, 255, 214} } //yellow and green
1218 };
1219 //take a set of 4 key colors from keyColors[3][4][3] and generate 16 colors
1220 // 328P will only have one set of full colors, Teensy will have two
1221 for (byte kcol = 0; kcol < 4; kcol++)
1222 {
1223 //go through each Key color
1224 const HSVColor* wc = &keyColors[colorPalNum][kcol];
1225 HSVColor workColor = {
1226 pgm_read_byte(&wc->h),
1227 pgm_read_byte(&wc->s),
1228 pgm_read_byte(&wc->v)
1229 };
1230 byte tnum = kcol + fTweens * kcol;
1231 HSVColor* tc = &fAllColors[tnum];
1232 tc->h = workColor.h;
1233 tc->s = workColor.s;
1234 //Value (V) is adjusted down to whatever brightness setting we've specified
1235 brightVal = min(brightVal, (byte)MAX_BRIGHTNESS);
1236 tc->v = map8(workColor.v, MIN_BRIGHTNESS, brightVal);
1237 if (tnum + 1 != fTotalColors)
1238 {
1239 const HSVColor* hsvptr = &keyColors[colorPalNum][kcol + 1];
1240 for (byte el = 0; el < 3; el++)
1241 {
1242 //loop through H, S and V from this key to the next
1243 int perStep = int(pgm_read_byte(&hsvptr->c[el]) - workColor.c[el]) / (fTweens + 1);
1244 if (perStep != 0)
1245 {
1246 for (byte tweenCount = 1; tweenCount <= fTweens; tweenCount++)
1247 {
1248 byte val = workColor.c[el] + tweenCount * perStep;
1249 tc[tweenCount].c[el] = (el == 2) ?
1250 map8(val, MIN_BRIGHTNESS, brightVal) : val;
1251 }
1252 }
1253 else
1254 {
1255 //tweens for this element (h,s or v) don't change between this key and the next, fill em up
1256 for (byte tweenCount = 1; tweenCount <= fTweens; tweenCount++)
1257 {
1258 tc[tweenCount].c[el] = (el == 2) ?
1259 map8(workColor.c[el], MIN_BRIGHTNESS, brightVal) : workColor.c[el];
1260 }
1261 }
1262 }
1263 }
1264 }
1265 }
