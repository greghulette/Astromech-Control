268 void driveStick(JoystickController* stick, float speedModifier)
269 {
270 fWasConnected = true;
271 if (!fEnabled)
272 {
273 stop();
274 }
275 else if (stick->state.button.l1)
276 {
277 if (!fMotorsStopped)
278 {
279 MOTOR_DEBUG_PRINTLN("STOP");
280 stop();
281 }
282 /* Disable Target Steering */
283 setTargetSteering(nullptr);
284 }
285 else
286 {
287 uint32_t currentMillis = millis();
288 if (currentMillis - fLastCommand > fSerialLatency)
289 {
290 // float drive_mod = speedModifier * -1.0f;
291 float drive_mod = throttleSpeed(speedModifier);
292 float turning = (float)(stick->state.analog.stick.lx + 128) / 127.5 - 1.0;
293 float throttle = (float)(stick->state.analog.stick.ly + 128) / 127.5 - 1.0;
294 if (fThrottleInverted)
295 throttle = -throttle;
296 if (fTurnInverted)
297 turning = -turning;
298
299 if (abs(turning) < 0.2)
300 turning = 0;
301 else
302 turning = pow(abs(turning)-0.2, 1.4) * ((turning < 0) ? -1 : 1);
303 if (abs(throttle) < 0.2)
304 throttle = 0;
305
306 if (fTargetSteering)
307 {
308 float targetThrottle = fTargetSteering->getThrottle();
309 float targetTurning = fTargetSteering->getTurning();
310 throttle = (throttle != 0) ? throttle : targetThrottle;
311 turning = (turning != 0) ? turning : targetTurning;
312 }
313
314 // clamp turning if throttle is greater than turning
315 // theory being that if you are at top speed turning should
316 // be less responsive. if you are full on turning throttle
317 // should remain responsive.
318 // if (abs(turning) <= abs(fDriveThrottle))
319 // {
320 // if (abs(fDriveThrottle) >= 0.8)
321 // turning /= 8;
322 // else if (abs(fDriveThrottle) >= 0.6)
323 // turning /= 6;
324 // else if (abs(fDriveThrottle) >= 0.4)
325 // turning /= 4;
326 // else if (abs(fDriveThrottle) >= 0.2)
327 // turning /= 2;
328 // }
329 if (turning != 0 || throttle != 0)
330 {
331 MOTOR_DEBUG_PRINT("TURNING "); MOTOR_DEBUG_PRINT(turning);
332 MOTOR_DEBUG_PRINT(" THROTTLE "); MOTOR_DEBUG_PRINTLN(throttle);
333 }
334 if (fScaling)
335 {
336 if (throttle > fDriveThrottle)
337 {
338 float scale = fThrottleAccelerationScale;
339 if (fDriveThrottle < 0)
340 {
341 MOTOR_DEBUG_PRINT("DECELERATING ");
342 scale = fThrottleDecelerationScale;
343 }
344 else
345 {
346 MOTOR_DEBUG_PRINT("ACCELERATING REVERSE ");
347 }
348 float val = max(abs(throttle - fDriveThrottle) / scale, 0.01f);
349 MOTOR_DEBUG_PRINT(val);
350 MOTOR_DEBUG_PRINT(" throttle: ");
351 MOTOR_DEBUG_PRINT(throttle);
352 MOTOR_DEBUG_PRINT(" drive : ");
353 MOTOR_DEBUG_PRINT(fDriveThrottle );
354 MOTOR_DEBUG_PRINT(" => ");
355 fDriveThrottle = ((int)round(min(fDriveThrottle + val, throttle)*100))/100.0f;
356 MOTOR_DEBUG_PRINTLN(fDriveThrottle );
357 }
358 else if (throttle < fDriveThrottle)
359 {
360 float scale = fThrottleAccelerationScale;
361 if (fDriveThrottle > 0)
362 {
363 MOTOR_DEBUG_PRINT("DECELERATING REVERSE ");
364 scale = fThrottleDecelerationScale;
365 }
366 else
367 {
368 MOTOR_DEBUG_PRINT("ACCELERATING ");
369 }
370 float val = abs(throttle - fDriveThrottle) / scale;
371 MOTOR_DEBUG_PRINT(val);
372 MOTOR_DEBUG_PRINT(" throttle: ");
373 MOTOR_DEBUG_PRINT(throttle);
374 MOTOR_DEBUG_PRINT(" drive : ");
375 MOTOR_DEBUG_PRINT(fDriveThrottle );
376 MOTOR_DEBUG_PRINT(" => ");
377 fDriveThrottle = ((int)floor(max(fDriveThrottle - val, throttle)*100))/100.0f;
378 MOTOR_DEBUG_PRINTLN(fDriveThrottle );
379 }
380 // Scale turning by fDriveThrottle
381 turning *= (1.0f - abs(fDriveThrottle) * 0.1);
382 if (turning > fDriveTurning)
383 {
384 float scale = fTurnAccelerationScale;
385 if (fDriveTurning < 0)
386 {
387 MOTOR_DEBUG_PRINT("DECELERATING LEFT ");
388 scale = fTurnDecelerationScale;
389 }
390 else
391 {
392 MOTOR_DEBUG_PRINT("ACCELERATING RIGHT ");
393 }
394 float val = max(abs(turning - fDriveTurning) / scale, 0.01f);
395 MOTOR_DEBUG_PRINT(val);
396 MOTOR_DEBUG_PRINT(" turning: ");
397 MOTOR_DEBUG_PRINT(turning);
398 MOTOR_DEBUG_PRINT(" drive : ");
399 MOTOR_DEBUG_PRINT(fDriveTurning );
400 MOTOR_DEBUG_PRINT(" => ");
401 fDriveTurning = ((int)round(min(fDriveTurning + val, turning)*100))/100.0f;
402 MOTOR_DEBUG_PRINTLN(fDriveTurning );
403 MOTOR_DEBUG_PRINT(" ");
404 }
405 else if (turning < fDriveTurning)
406 {
407 float scale = fTurnAccelerationScale;
408 if (fDriveTurning > 0)
409 {
410 MOTOR_DEBUG_PRINT("DECELERATING RIGHT ");
411 scale = fTurnDecelerationScale;
412 }
413 else
414 {
415 MOTOR_DEBUG_PRINT("ACCELERATING LEFT ");
416 }
417 float val = abs(turning - fDriveTurning) / scale;
418 MOTOR_DEBUG_PRINT(val);
419 MOTOR_DEBUG_PRINT(" turning: ");
420 MOTOR_DEBUG_PRINT(turning);
421 MOTOR_DEBUG_PRINT(" drive : ");
422 MOTOR_DEBUG_PRINT(fDriveTurning );
423 MOTOR_DEBUG_PRINT(" => ");
424 fDriveTurning = ((int)floor(max(fDriveTurning - val, turning)*100))/100.0f;
425 MOTOR_DEBUG_PRINTLN(fDriveTurning );
426 }
427 }
428 else
429 {
430 fDriveThrottle = throttle;
431 fDriveTurning = turning;
432 }
433
434 float x = fDriveThrottle;
435 float y = fDriveTurning;
436
437 float target_left = x;
438 float target_right = y;
439 if (fChannelMixing)
440 {
441 float r = hypot(x, y);
442 float t = atan2(y, x);
443 // rotate 45 degrees
444 t += M_PI / 4;
445
446 target_left = r * cos(t);
447 target_right = r * sin(t);
448
449 // rescale the new coords
450 target_left *= sqrt(2);
451 target_right *= sqrt(2);
452 }
453 // clamp to -1/+1 and apply max speed limit
454 target_left = max(-1.0f, min(target_left, 1.0f)) * drive_mod;
455 target_right = max(-1.0f, min(target_right, 1.0f)) * drive_mod;
456
457 motor(target_left, target_right);
458 fLastCommand = currentMillis;
459 fMotorsStopped = false;
460 }
