360 /* Send out a single command to the device */
361 void spiTransfer(byte device, volatile byte opcode, volatile byte data)
362 {
363 //Create an array with the data to shift out
364 int offset = device * 2;
365 for (byte i = 0; i < sizeof(fSPIData); i++)
366 fSPIData[i] = (byte)0;
367 //put our device data into the array
368 fSPIData[offset + 1] = opcode;
369 fSPIData[offset] = data;
370 //enable the line
371 digitalWrite(fSPI_CS, LOW);
372 //Now shift out the data
373 for (byte i = sizeof(fSPIData); i > 0; i--)
374 shiftOut(fSPI_MOSI, fSPI_CLK, MSBFIRST, fSPIData[i - 1]);
375 //latch the data onto the display
376 digitalWrite(fSPI_CS, HIGH);
377 }
