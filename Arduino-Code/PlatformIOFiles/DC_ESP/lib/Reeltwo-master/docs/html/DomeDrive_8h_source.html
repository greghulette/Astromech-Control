187 int shortestDistance(int origin, int target)
188 {
189 int result = 0.0;
190 int diff = fmod(fmod(abs(origin - target), 360), 360);
191
192 if (diff > 180)
193 {
194 //There is a shorter path in opposite direction
195 result = (360 - diff);
196 if (target > origin)
197 result *= -1;
198 }
199 else
200 {
201 result = diff;
202 if (origin > target)
203 result *= -1;
204 }
205 return result;
