278template<typename T, unsigned SIZE, typename CT>
279class FourierTransform
280{
281public:
282 FourierTransform(bool inverseTransform = false) :
283 inverse(inverseTransform)
284 {
285 unsigned n = 1 << nbits;
286 for (unsigned i = 0; i < n; i++)
287 rev[-split_radix_permutation(i, n, inverse) & (n - 1)] = i;
288 }
289
290 void permute(Complex<T>* data)
291 {
292 unsigned np = 1 << nbits;
293 for (unsigned i = 0; i < np; i++)
294 tmp[rev[i]] = data[i];
295 for (unsigned i = 0; i < np; i++)
296 data[i] = tmp[i];
297 }
298
299 void calculate(Complex<T>* data)
300 {
301 dispatch(nbits)(data, nbits);
302 }
303
304private:
305 typedef void (*FFT)(Complex<T>* data, unsigned nbits);
306 static constexpr int nbits = log2(SIZE);
307 int inverse;
308 int mdct_size;
309 int mdct_bits;
310 unsigned rev[SIZE];
311 Complex<T> tmp[SIZE];
312 const T* tcos = CT::get(nbits);
313 const T* tsin = tcos + (SIZE >> 2);
314
315 static int split_radix_permutation(int i, int n, int inverse)
316 {
317 int m;
318 if (n <= 2)
319 return i & 1;
320 m = n >> 1;
321 if (!(i&m))
322 return split_radix_permutation(i, m, inverse) * 2;
323 m >>= 1;
324 if (inverse == !(i&m))
325 return split_radix_permutation(i, m, inverse) * 4 + 1;
326 return split_radix_permutation(i, m, inverse) * 4 - 1;
327 }
328
329 #define BF(x, y, a, b) do { \
330 x = a - b; \
331 y = a + b; \
332 } while (0)
333
334 #define CMUL(dre, dim, are, aim, bre, bim) do { \
335 (dre) = (are) * (bre) - (aim) * (bim); \
336 (dim) = (are) * (bim) + (aim) * (bre); \
337 } while (0)
338
339 #define BUTTERFLIES(a0,a1,a2,a3) {\
340 BF(t3, t5, t5, t1);\
341 BF(a2.real, a0.real, a0.real, t5);\
342 BF(a3.img, a1.img, a1.img, t3);\
343 BF(t4, t6, t2, t6);\
344 BF(a3.real, a1.real, a1.real, t4);\
345 BF(a2.img, a0.img, a0.img, t6);\
346 }
347
348 // force loading all the inputs before storing any.
349 // this is slightly slower for small data, but avoids store->load aliasing
350 // for addresses separated by large powers of 2.
351 #define BUTTERFLIES_BIG(a0,a1,a2,a3) {\
352 T r0=a0.real, i0=a0.img, r1=a1.real, i1=a1.img;\
353 BF(t3, t5, t5, t1);\
354 BF(a2.real, a0.real, r0, t5);\
355 BF(a3.img, a1.img, i1, t3);\
356 BF(t4, t6, t2, t6);\
357 BF(a3.real, a1.real, r1, t4);\
358 BF(a2.img, a0.img, i0, t6);\
359 }
360
361 #define TRANSFORM(a0,a1,a2,a3,wre,wim) {\
362 CMUL(t1, t2, a2.real, a2.img, wre, -wim);\
363 CMUL(t5, t6, a3.real, a3.img, wre, wim);\
364 BUTTERFLIES(a0,a1,a2,a3)\
365 }
366
367 #define TRANSFORM_ZERO(a0,a1,a2,a3) {\
368 t1 = a2.real;\
369 t2 = a2.img;\
370 t5 = a3.real;\
371 t6 = a3.img;\
372 BUTTERFLIES(a0,a1,a2,a3)\
373 }

375 static void fft4(Complex<T>* z, unsigned n)
376 {
377 // argument not used
378 (void)n;
379
380 T t1, t2, t3, t4, t5, t6, t7, t8;
381
382 BF(t3, t1, z[0].real, z[1].real);
383 BF(t8, t6, z[3].real, z[2].real);
384 BF(z[2].real, z[0].real, t1, t6);
385 BF(t4, t2, z[0].img, z[1].img);
386 BF(t7, t5, z[2].img, z[3].img);
387 BF(z[3].img, z[1].img, t4, t8);
388 BF(z[3].real, z[1].real, t3, t7);
389 BF(z[2].img, z[0].img, t2, t5);
390 }

392 static void fft8(Complex<T>* z, unsigned n)
393 {
394 // argument not used
395 (void)n;
396
397 T t1, t2, t3, t4, t5, t6;
398
399 fft4(z, 2);
400
401 BF(t1, z[5].real, z[4].real, -z[5].real);
402 BF(t2, z[5].img, z[4].img, -z[5].img);
403 BF(t5, z[7].real, z[6].real, -z[7].real);
404 BF(t6, z[7].img, z[6].img, -z[7].img);
405
406 BUTTERFLIES(z[0],z[2],z[4],z[6]);
407 TRANSFORM(z[1],z[3],z[5],z[7],sqrt1_2,sqrt1_2);
408 }

410 static void fft16(Complex<T>* z, unsigned n)
411 {
412 // argument not used
413 (void)n;
414
415 T t1, t2, t3, t4, t5, t6;
416 const T* cos16 = CT::get(4);
417 T cos_16_1 = READ_COS(cos16[1]);
418 T cos_16_3 = READ_COS(cos16[3]);
419
420 fft8(z, 3);
421 fft4(z+8, 2);
422 fft4(z+12, 2);
423
424 TRANSFORM_ZERO(z[0],z[4],z[8],z[12]);
425 TRANSFORM(z[2],z[6],z[10],z[14],sqrt1_2,sqrt1_2);
426 TRANSFORM(z[1],z[5],z[9],z[13],cos_16_1,cos_16_3);
427 TRANSFORM(z[3],z[7],z[11],z[15],cos_16_3,cos_16_1);
428 }

430 static void pass(Complex<T> *z, const T* wre, unsigned int n)
431 {
432 T t1, t2, t3, t4, t5, t6;
433 int o1 = 2*n;
434 int o2 = 4*n;
435 int o3 = 6*n;
436 const T* wim = wre+o1;
437
438 n--;
439 TRANSFORM_ZERO(z[0],z[o1],z[o2],z[o3]);
440 TRANSFORM(z[1],z[o1+1],z[o2+1],z[o3+1],READ_COS(wre[1]),READ_COS(wim[-1]));
441 do
442 {
443 z += 2;
444 wre += 2;
445 wim -= 2;
446 TRANSFORM(z[0],z[o1],z[o2],z[o3],READ_COS(wre[0]),READ_COS(wim[0]));
447 TRANSFORM(z[1],z[o1+1],z[o2+1],z[o3+1],READ_COS(wre[1]),READ_COS(wim[-1]));
448 }
449 while(--n);
450 }

479 static void fft(Complex<T>* z, unsigned n)
480 {
481 unsigned n4 = (1<<(n-2));
482 dispatch(n-1)(z, n-1);
483 dispatch(n-2)(z+n4*2, n-2);
484 dispatch(n-2)(z+n4*3, n-2);
485 #ifndef CTFFT_MINIMAL
486 if (n >= 10)
487 passBig(z, CT::get(n), n4/2);
488 else
489 #endif
490 pass(z, CT::get(n), n4/2);
491 }

493 static FFT dispatch(unsigned n)
494 {
495 static FFT opt[] = {
496 fft4,
497 fft8,
498 fft16
499 };
500 return (n-2 < sizeof(opt)/sizeof(opt[0])) ? opt[n-2] : fft;
501 }

511template<typename T, unsigned SIZE>
512class RealDiscrete
513{
514public:
515 enum TransformType
516 {
517 kR2C,
518 kC2R,
519 kInverseR2C,
520 kInverseC2R
521 };
522
523 RealDiscrete(TransformType transform = kR2C) :
524 inverse(transform == kC2R || transform == kC2R),
525 negative(transform == kC2R || transform == kR2C),
526 sign_convention((transform == kInverseR2C || transform == kC2R) ? 1 : -1),
527 fft(inverse)
528 {
529 }
530
531 void calculate(T* data)
532 {
533 Complex<T> ev, od, odsum;
534 const int n = 1 << nbits;
535 const float k1 = 0.5;
536 const float k2 = 0.5 - inverse;
537 const T* tcos = this->tcos;
538 const T* tsin = this->tsin;
539
540 if (!inverse)
541 {
542 fft.permute((Complex<T>*)data);
543 fft.calculate((Complex<T>*)data);
544 }
545
546 /* i=0 is a special case because of packing, the DC term is real, so we
547 are going to throw the N/2 term (also real) in with it. */
548 ev.real = data[0];
549 data[0] = ev.real + data[1];
550 data[1] = ev.real - data[1];
551
552 int i;
553 #define RDFT_UNMANGLE(sign0, sign1) \
554 for (i = 1; i < (n>>2); i++) \
555 { \
556 int i1 = 2*i; \
557 int i2 = n-i1; \
558 /* Separate even and odd FFTs */ \
559 ev.real = k1*(data[i1 ]+data[i2 ]); \
560 od.img = k2*(data[i2 ]-data[i1 ]); \
561 ev.img = k1*(data[i1+1]-data[i2+1]); \
562 od.real = k2*(data[i1+1]+data[i2+1]); \
563 /* Apply twiddle factors to the odd FFT and add to the even FFT */ \
564 odsum.real = od.real* READ_COS(tcos[i]) sign0 od.img * READ_COS(tsin[i]);\
565 odsum.img = od.img * READ_COS(tcos[i]) sign1 od.real* READ_COS(tsin[i]);\
566 data[i1 ] = ev.real + odsum.real; \
567 data[i1+1] = ev.img + odsum.img; \
568 data[i2 ] = ev.real - odsum.real; \
569 data[i2+1] = odsum.img - ev.img; \
570 }
571
572 if (negative)
573 {
574 RDFT_UNMANGLE(+,-)
575 }
576 else
577 {
578 RDFT_UNMANGLE(-,+)
579 }
580 #undef RDFT_UNMANGLE
581
582 data[2*i+1] = sign_convention * data[2*i+1];
583 if (inverse)
584 {
585 data[0] *= k1;
586 data[1] *= k1;
587 fft.permute((Complex<T>*)data);
588 fft.calculate((Complex<T>*)data);
589 }
590 }
591
592 static constexpr unsigned nbits = log2(SIZE);
593 using costable = CosTable<T, nbits>;
594 bool inverse;
595 bool negative;
596 int sign_convention;
597 FourierTransform<T, 1<<(nbits-1), costable> fft;
598 const T* tcos = costable::get(nbits);
599 const T* tsin = tcos + (SIZE >> 2);
600};
