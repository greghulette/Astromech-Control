409 void setPrescale(uint8_t prescale)
410 {
411 for (byte chip = 0; chip < numberOfPCA9685Chips(); chip++)
412 {
413 uint8_t currentMode = readRegister(chip, PCA9685_MODE1); //read current MODE1 register
414 uint8_t sleepMode = (currentMode & 0x7F) | PCA9685_SLEEP;
415 writeRegister(chip, PCA9685_MODE1, sleepMode);
416 writeRegister(chip, PCA9685_PRESCALE, prescale); //change the prescaler register
417 writeRegister(chip, PCA9685_MODE1, currentMode);
418 }

468 void setUpdateFrequency(const float targetUpdateFrequency[(numServos/16)+1])
469 {
470 for (byte chip = 0; chip < numberOfPCA9685Chips(); chip++)
471 {
472 fTargetUpdateFrequency[chip] = targetUpdateFrequency[chip];
473 SERVO_DEBUG_PRINT("Set target update frequency[");
474 SERVO_DEBUG_PRINT(chip);
475 SERVO_DEBUG_PRINT("] ");
476 SERVO_DEBUG_PRINT(targetUpdateFrequency[chip]);
477 SERVO_DEBUG_PRINTLN();
478
479 uint8_t prescale = getCalculatedPrescale(fClocks[chip], fTargetUpdateFrequency[chip]);
480
481 uint8_t currentMode;
482 uint8_t sleepMode;
483
484 currentMode = readRegister(chip, PCA9685_MODE1); //read current MODE1 register
485 sleepMode = (currentMode & 0x7F) | PCA9685_SLEEP;
486 writeRegister(chip, PCA9685_MODE1, sleepMode);
487 writeRegister(chip, PCA9685_PRESCALE, prescale); //change the prescaler register
488 writeRegister(chip, PCA9685_MODE1, currentMode);
489
490 delay(1);
491
492 SERVO_DEBUG_PRINT("Set prescale[");
493 SERVO_DEBUG_PRINT(chip);
494 SERVO_DEBUG_PRINT("] : ");
495 SERVO_DEBUG_PRINTLN(prescale);
496
497 prescale = getPrescale(chip);
498 fCalculatedUpdateFrequency[chip] = fClocks[chip] / (float)(((uint32_t)prescale + 1) << 12);
499 fCalculatedResolution[chip] = 1000000.0 / (fCalculatedUpdateFrequency[chip] * 4096);
500
501 SERVO_DEBUG_PRINT("Calculated update frequency = ");
502 SERVO_DEBUG_PRINTLN(fCalculatedUpdateFrequency[chip]);
503 }

506 uint8_t getCalculatedPrescale(uint32_t calculatedClockFrequency, float targetUpdateFrequency)
507 {
508 //rounding method from here: https://github.com/adafruit/Adafruit-PWM-Servo-Driver-Library/issues/40
509 uint32_t calculatedPrescale;
510 calculatedPrescale = (float)calculatedClockFrequency * 100.0 / (4096.0 * targetUpdateFrequency);
511 if ((calculatedPrescale % 100) >= 50)
512 calculatedPrescale += 100;
513 calculatedPrescale = (calculatedPrescale / 100) - 1;
514 return (uint8_t)calculatedPrescale;
515 }

568 void move(ServoDispatchPCA9685<numServos,defaultOEValue>* dispatch, uint32_t timeNow)
569 {
570 if (finishTime != 0)
571 {
572 if (timeNow < startTime)
573 {
574 /* wait */
575 }
576 else if (timeNow >= finishTime)
577 {
578 posNow = finishPos;
579 doMove(dispatch, timeNow);
580 init();
581 }
582 else if (lastMoveTime != timeNow)
583 {
584 uint32_t timeSinceLastMove = timeNow - startTime;
585 uint32_t denominator = finishTime - startTime;
586 float (*useMethod)(float) = easingMethod;
587 if (useMethod == NULL)
588 useMethod = Easing::LinearInterpolation;
589 float fractionChange = useMethod(float(timeSinceLastMove)/float(denominator));
590 int distanceToMove = float(deltaPos) * fractionChange;
591 uint16_t newPos = startPosition + distanceToMove;
592 if (newPos != posNow)
593 {
594 posNow = startPosition + distanceToMove;
595 doMove(dispatch, timeNow);
596 }
597 }
598 }
599 }

601 void moveToPulse(ServoDispatchPCA9685<numServos,defaultOEValue>* dispatch, uint32_t startDelay, uint32_t moveTime, uint16_t startPos, uint16_t pos)
602 {
603 uint32_t timeNow = millis();
604
605 startTime = startDelay + timeNow;
606 finishTime = moveTime + startTime;
607 finishPos = min(getMaximum(), max(getMinimum(), pos));
608 posNow = startPosition = startPos;
609 deltaPos = finishPos - posNow;
610 doMove(dispatch, timeNow);
611 }

702 virtual void _moveServoSetToPulse(uint32_t servoGroupMask, uint32_t servoSetMask, uint32_t startDelay, uint32_t moveTimeMin, uint32_t moveTimeMax, uint16_t onPos, uint16_t offPos) override
703 {
704 byte bitShift = 31;
705 for (uint16_t i = 0; i < numServos; i++)
706 {
707 if ((fServos[i].group & servoGroupMask) == 0)
708 continue;
709 uint32_t moveTime = (moveTimeMin != moveTimeMax) ? random(moveTimeMin, moveTimeMax) : moveTimeMax;
710 bool on = ((servoSetMask & (1L<<bitShift)) != 0);
711 moveToPulse(i, startDelay, moveTime, fServos[i].currentPos(), (on) ? onPos : offPos);
712 if (bitShift-- == 0)
713 break;
714 }
